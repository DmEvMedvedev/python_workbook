#Таблица 2.14. Таблица рейтингов
#Рейтинг
#0,0 Низкий уровень
#0,4 Удовлетворительный уровень
#0,6 и выше Высокий уровень
#Напишите программу, которая будет запрашивать у пользователя рейтинг
#сотрудника и выводить соответствующее значение из приведенной таб­лицы.
#Также необходимо показать сумму прибавки сотрудника. При вводе некор-
#ректного значения рейтинга программа должна об этом сообщать.

rating=float(input("Employee rating"))
increase="None"
rate=2400

    rating_class="Low level"
    increase=rate*rating

    rating_class="High level"
